- 나중에 뭐가 틀렸는지 찾아보기(idea는 가로세로를 바꾼 pair를 하나 더 만들어서 붙어있는걸 검사)
~~~cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main(){
    int T;
    scanf("%d", &T);
    for(int i=0;i<T;i++){
        int M, N, K;
        scanf("%d %d %d", &M, &N, &K);
        if(M+N<3 && K>0){
            printf("%d", 1);
            return 0;
        }
        pair<int, int> p;
        vector<pair<int,int>> v;
        vector<pair<int,int>> v1;
        for(int i=0;i<K;i++){
            scanf("%d %d", &p.first, &p.second);
            v.push_back(p);
            v1.push_back(make_pair(p.second, p.first));
        }
        sort(v.begin(), v.end());
        sort(v1.begin(), v1.end());
        int cnt=1;
        for(int i=0;i<K;i++){
            //cout<<'['<<v[i].first<<','<<v[i].second<<']'<<'\n';
            //cout<<'['<<v1[i].first<<','<<v1[i].second<<']'<<'\n';
            if(v[i].first==v[i+1].first||v1[i].first==v1[i+1].first) {
                if(v[i].first==v[i+1].first){
                    if(v[i].second-v[i+1].second ==1 ||v[i].second-v[i+1].second ==-1)
                        continue;
                    else cnt++;
                    
                }
                else if(v1[i].first==v1[i+1].first){
                    if(v1[i].second-v1[i+1].second==1 ||v1[i].second-v1[i+1].second==-1)
                        continue;
                    else cnt++;
                }
                else {
                    cnt++;
                    //cout<<"i >> "<<i<<'\n';
                }
            }
            else {
                cnt++;
                //cout<<"else "<<i<<'\n';
            }
        }
        printf("%d", cnt-1);
    }
    

    return 0;
}
~~~
